

    \filetitle{empty}{Empty tseries object preserving its size in 2nd and higher dimensions}{tseries/empty}

	\paragraph{Syntax}\label{syntax}

\begin{verbatim}
x = empty(x)
\end{verbatim}

\paragraph{Input arguments}\label{input-arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{x} {[} tseries {]} - Tseries object that will be emptied.
\end{itemize}

\paragraph{Output arguments}\label{output-arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{x} {[} tseries {]} - Empty tseries object with the 2nd and
  higher dimensions the same size as the input tseries object, and
  comments preserved.
\end{itemize}
