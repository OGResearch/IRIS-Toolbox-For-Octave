

    \filetitle{dat2ypf}{Convert IRIS serial date number to year, period and frequency}{dates/dat2ypf}

	\paragraph{Syntax}\label{syntax}

\begin{verbatim}
[Y,P,F] = dat2ypf(Dat)
\end{verbatim}

\paragraph{Input arguments}\label{input-arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{Dat} {[} numeric {]} - IRIS serial date numbers.
\end{itemize}

\paragraph{Output arguments}\label{output-arguments}

\begin{itemize}
\item
  \texttt{Y} {[} numeric {]} - Years.
\item
  \texttt{P} {[} numeric {]} - Periods within year.
\item
  \texttt{F} {[} numeric {]} - Date frequencies.
\end{itemize}
