

    \filetitle{mm}{IRIS serial date numbers for dates with monthly frequency}{dates/mm}

	\paragraph{Syntax}\label{syntax}

\begin{verbatim}
d = mm(y)
d = mm(y,m)
\end{verbatim}

\paragraph{Input arguments}\label{input-arguments}

\begin{itemize}
\item
  \texttt{y} {[} numeric {]} - Years.
\item
  \texttt{m} {[} numeric {]} - Months; if missing, first month (January)
  is assumed.
\end{itemize}

\paragraph{Output arguments}\label{output-arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{d} {[} numeric {]} - IRIS serial date numbers representing the
  input months.
\end{itemize}
