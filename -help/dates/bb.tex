

    \filetitle{bb}{IRIS serial date numbers for dates with bi-monthly frequency}{dates/bb}

	\paragraph{Syntax}\label{syntax}

\begin{verbatim}
d = bb(y)
d = bb(y,b)
\end{verbatim}

\paragraph{Input arguments}\label{input-arguments}

\begin{itemize}
\item
  \texttt{y} {[} numeric {]} - Years.
\item
  \texttt{q} {[} numeric {]} - B-months; if missing, first bi-month is
  assumed.
\end{itemize}

\paragraph{Output arguments}\label{output-arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{d} {[} numeric {]} - IRIS serial date numbers representing the
  input bi-months.
\end{itemize}
