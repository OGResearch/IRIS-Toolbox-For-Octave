

    \filetitle{dd}{Matlab serial date numbers that can be used to construct daily tseries objects}{dates/dd}

	\paragraph{Syntax}\label{syntax}

\begin{verbatim}
Dat = dd(Year,Month,Day)
\end{verbatim}

\paragraph{Output arguments}\label{output-arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{Dat} {[} numeric {]} - IRIS serial date numbers.
\end{itemize}

\paragraph{Input arguments}\label{input-arguments}

\begin{itemize}
\item
  \texttt{Year} {[} numeric {]} - Years.
\item
  \texttt{Month} {[} numeric {]} - Months.
\item
  \texttt{Day} {[} numeric {]} - Days.
\end{itemize}
