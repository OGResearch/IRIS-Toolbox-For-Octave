

    \filetitle{datbom}{Beginning of month for the specified daily date}{dates/datbom}

	\paragraph{Syntax}\label{syntax}

\begin{verbatim}
Bom = datebom(D)
\end{verbatim}

\paragraph{Input arguments}\label{input-arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{D} {[} numeric {]} - Daily serial date number.
\end{itemize}

\paragraph{Output arguments}\label{output-arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{Bom} {[} numeric {]} - Daily serial date number for the first
  day of the same month as \texttt{D}.
\end{itemize}
