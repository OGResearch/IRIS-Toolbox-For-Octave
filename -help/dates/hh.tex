

    \filetitle{hh}{IRIS serial date numbers for dates with half-yearly frequency}{dates/hh}

	\paragraph{Syntax}\label{syntax}

\begin{verbatim}
d = hh(y)
d = hh(y,h)
\end{verbatim}

\paragraph{Input arguments}\label{input-arguments}

\begin{itemize}
\item
  \texttt{y} {[} numeric {]} - Years.
\item
  \texttt{h} {[} numeric {]} - Half-years; if missing, first half-year
  is used.
\end{itemize}

\paragraph{Output arguments}\label{output-arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{d} {[} numeric {]} - IRIS serial date numbers representing the
  input half-years.
\end{itemize}
