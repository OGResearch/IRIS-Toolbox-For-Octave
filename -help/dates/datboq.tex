

    \filetitle{datboq}{Beginning of quarter for the specified daily date}{dates/datboq}

	\paragraph{Syntax}\label{syntax}

\begin{verbatim}
Boq = datboq(D)
\end{verbatim}

\paragraph{Input arguments}\label{input-arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{D} {[} numeric {]} - Daily serial date number.
\end{itemize}

\paragraph{Output arguments}\label{output-arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{Boq} {[} numeric {]} - Daily serial date number for the first
  day of the same quarter as \texttt{D}.
\end{itemize}
