

    \filetitle{dat2dec}{Convert dates to their decimal representations}{dates/dat2dec}

	\paragraph{Syntax}\label{syntax}

\begin{verbatim}
DEC = dat2dec(DAT)
\end{verbatim}

\paragraph{Input arguments}\label{input-arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{DAT} {[} numeric {]} - IRIS serial date number.
\end{itemize}

\paragraph{Output arguments}\label{output-arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{DEC} {[} numeric {]} - Decimal number representing the input
  dates, computed as \texttt{year + (per-1)/freq}.
\end{itemize}
