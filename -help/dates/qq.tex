

    \filetitle{qq}{IRIS serial date numbers for dates with quarterly frequency}{dates/qq}

	\paragraph{Syntax}\label{syntax}

\begin{verbatim}
d = qq(y)
d = qq(y,q)
\end{verbatim}

\paragraph{Input arguments}\label{input-arguments}

\begin{itemize}
\item
  \texttt{y} {[} numeric {]} - Years.
\item
  \texttt{q} {[} numeric {]} - Quarters; if missing, first quarter is
  assumed.
\end{itemize}

\paragraph{Output arguments}\label{output-arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{d} {[} numeric {]} - IRIS serial date numbers representing the
  input quarterly dates.
\end{itemize}
