

    \filetitle{datboy}{Beginning of year for the specified daily date}{dates/datboy}

	\paragraph{Syntax}\label{syntax}

\begin{verbatim}
Boy = dateboy(D)
\end{verbatim}

\paragraph{Input arguments}\label{input-arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{D} {[} numeric {]} - Daily serial date number.
\end{itemize}

\paragraph{Output arguments}\label{output-arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{Boy} {[} numeric {]} - Daily serial date number for the first
  day of the same year as \texttt{D}.
\end{itemize}
